### 1，将1~100之间的所有正整数存放在一个```List```集合中，并将集合中索引位置是10的对象从集合中移除。

		// TODO 自动生成的方法存根

		for(int i=1;i<=100;i++)
			list.add(i);
		
		for(int i=0;i<list.size();i++) {
			if(i%10==0&&i!=0)
				out.println(list.get(i));
			else
				out.print(list.get(i)+ " ");
		}
		out.println("");
		out.println("共"+list.size()+"个元素");
		
		out.println("移除之后：");
		list.remove(10);

			else
				out.print(list.get(i)+ " ");
		}
		out.println("");
		out.println("共"+list.size()+"个元素");
	}

}
